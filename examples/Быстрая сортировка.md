# Быстрая сортировка

```
//Алгоритм "Быстрая сортировка"
Функция БыстраяСортировка(Знач Массив) Экспорт    
    НижняяГраница = 0;
    ВерхняяГраница = Массив.ВГраница();    
    б_Сортировка(Массив,НижняяГраница,ВерхняяГраница);    
    Возврат Массив;    
КонецФункции // БыстраяСортировка()

Процедура б_Сортировка(Массив,НижнийПредел,ВерхнийПредел)    
    i    = НижнийПредел;
    j    = ВерхнийПредел;
    m    = Массив[Цел((i+j)/2)];    
    Пока i<=j Цикл        
        Пока Массив[i] < m Цикл            
            i	= i + 1;                   
        КонецЦикла;        
        Пока Массив[j] > m Цикл            
            j	= j - 1;                   
        КонецЦикла;
		Если i <= j Тогда               
            Замена       = Массив[i];
            Массив[i]    = Массив[j];
            Массив[j]    = Замена;
            i            = i + 1;
            j            = j - 1;            
        КонецЕсли;        
		//Если i > j Тогда                       
		//    Прервать;                        
		//КонецЕсли;
    КонецЦикла;    
    Если НижнийПредел < j Тогда         
        б_Сортировка(Массив,НижнийПредел,j);        
    КонецЕсли;    
    Если i < ВерхнийПредел Тогда                      
        б_Сортировка(Массив,i,ВерхнийПредел);        
    КонецЕсли;    
КонецПроцедуры // б_Сортировка()
```
